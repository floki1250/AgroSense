@startuml
title Add Item to Inventory
hide footbox
actor Manager
activate Manager
Manager -> InventoryView : Authentication Request
alt Successful Authentication
    InventoryView --> Manager : Authentication Accepted
    Manager -> InventoryView : Add Item
    activate InventoryView
    InventoryView --> Manager : Show Add Item Form
    Manager -> InventoryView : Fill the Form
    InventoryView -> InventoryController : Send Data
    activate InventoryController
    deactivate InventoryView
    InventoryController -> InventoryModel : Send Data
    activate InventoryModel
    deactivate InventoryController
    InventoryModel -> Database : Save Item Data Request
    deactivate InventoryModel
    activate Database
    Database -> Database : Save data
    alt Data Saved Successfully
        Database --> InventoryController : Successful
        InventoryController --> InventoryView : Successful
        deactivate InventoryController
        InventoryView --> Manager : Successful
        deactivate Manager
    else Data Save Failed
        Database --> InventoryController : Failed
        deactivate Database
        InventoryController --> InventoryView : Failed
        deactivate InventoryController
        InventoryView --> Manager : Data save failed
        deactivate Manager
    end
else Authentication Failed
    InventoryView --> Manager : Please repeat
end

@enduml
