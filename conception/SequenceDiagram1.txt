@startuml
title Authentication
autonumber 1
actor User
activate User
User -> AuthenticationView: Login request
activate AuthenticationView
AuthenticationView -> User: Login form
deactivate AuthenticationView
User -> AuthenticationView: Submit login form
activate AuthenticationView
AuthenticationView -> AuthenticationController: Authentication request
activate AuthenticationController
deactivate AuthenticationView
AuthenticationController -> UserModel: Authenticate user data
activate UserModel
deactivate AuthenticationController
UserModel -> Database: Authenticate user data
activate Database 
deactivate UserModel
Database -> Database : Verify Data 

alt Valid Credentials
    Database -> UserModel: Notify success
    UserModel -> AuthenticationController: Notify success
    AuthenticationController -> AuthenticationView: Notify success
    AuthenticationView -> User: Display success message
else Invalid Credentials
    Database -> UserModel: Notify failure
    UserModel -> AuthenticationController: Notify failure
    AuthenticationController -> AuthenticationView: Notify failure
    AuthenticationView -> User: Display failure message
end

@enduml
